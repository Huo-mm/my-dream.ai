// 狀態管理：目前選中的角色 ID let currentId = localStorage.getItem('active_char_id') || 'default'; let chatHistory = []; let characters = JSON.parse(localStorage.getItem('char_configs')) || { 'default': { name: "AI 助手", prompt: "你是一個親切的助手喵。", history: [] } }; let activeModel = ""; function loadSettings() { const char = characters[currentId]; document.getElementById('apiKey').value = localStorage.getItem('my_ai_key') || ""; document.getElementById('charName').value = char.name; document.getElementById('charPrompt').value = char.prompt; document.getElementById('display-name').innerText = char.name; chatHistory = char.history || []; renderCharList(); // 渲染側邊欄的角色清單 } // --- 新增：在側邊欄顯示角色清單 --- function renderCharList() { let listHTML = ``; // 插入到 API Key 設定上方 const sidebar = document.getElementById('sidebar'); const oldSelector = document.getElementById('charSelectorArea'); if(oldSelector) oldSelector.remove(); const div = document.createElement('div'); div.id = 'charSelectorArea'; div.innerHTML = listHTML; sidebar.insertBefore(div, sidebar.firstChild.nextSibling); } function switchChar(id) { currentId = id; localStorage.setItem('active_char_id', id); location.reload(); } function addNewChar() { const name = prompt("請輸入新角色的名字：", "新朋友"); if(!name) return; const id = 'char_' + Date.now(); characters[id] = { name: name, prompt: "你是一個新的角色。", history: [] }; saveToLocalStorage(); switchChar(id); } function saveSettings() { localStorage.setItem('my_ai_key', document.getElementById('apiKey').value.trim()); // 更新目前角色的設定 characters[currentId].name = document.getElementById('charName').value.trim(); characters[currentId].prompt = document.getElementById('charPrompt').value.trim(); saveToLocalStorage(); alert("角色設定已更新！"); location.reload(); } function saveToLocalStorage() { localStorage.setItem('char_configs', JSON.stringify(characters)); } // 修改清除紀錄：針對當前角色 function clearHistory() { if(confirm("要將此角色的對話存入「記憶倉庫」並清空嗎？")) { if(chatHistory.length > 0) { const textContent = chatHistory.map(m => `${m.role === 'user' ? '使用者' : characters[currentId].name}: ${m.parts[0].text}`).join('\n'); const archives = JSON.parse(localStorage.getItem('chat_archives') || "[]"); archives.unshift({ date: new Date().toLocaleString() + ` (${characters[currentId].name})`, content: textContent }); localStorage.setItem('chat_archives', JSON.stringify(archives)); } characters[currentId].history = []; saveToLocalStorage(); location.reload(); } } async function initModel() { const key = localStorage.getItem('my_ai_key'); if(!key) return; try { const res = await fetch(`https://generativelanguage.googleapis.com/v1beta/models?key=${key}`); const data = await res.json(); if (data.models) { const found = data.models.find(m => m.name.includes('flash')) || data.models[0]; activeModel = found.name; } } catch (e) {} } function renderMessages() { const box = document.getElementById('chat-box'); box.innerHTML = ""; chatHistory.forEach(msg => addBubble(msg.parts[0].text, msg.role === 'user' ? 'u' : 'a', false)); box.scrollTop = box.scrollHeight; } async function send() { const key = localStorage.getItem('my_ai_key'); if(!key || !activeModel) { alert("請先設定 API Key"); return; } const input = document.getElementById('userInput'); const text = input.value.trim(); if(!text) return; addBubble(text, 'u'); input.value = ''; const userMsg = { role: "user", parts: [{ text: text }] }; // 讀取備忘錄、鬧鐘 const userPrefs = JSON.parse(localStorage.getItem('user_prefs') || "[]").join('、') || "無紀錄"; const userTodos = JSON.parse(localStorage.getItem('user_todos') || "[]").join('、') || "目前沒事做"; const savedAlarms = localStorage.getItem('my_alarms') || "無鬧鐘"; const now = new Date(); const systemContext = ` 【當前個人化環境】時間：${now.toLocaleString()} | 喜好：${userPrefs} | 待辦：${userTodos} | 鬧鐘：${savedAlarms} 【角色靈魂設定】${characters[currentId].prompt} `; try { const response = await fetch(`https://generativelanguage.googleapis.com/v1beta/${activeModel}:generateContent?key=${key}`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ system_instruction: { parts: [{ text: systemContext }] }, contents: [...chatHistory, userMsg] }) }); const data = await response.json(); const aiText = data.candidates[0].content.parts[0].text; addBubble(aiText, 'a'); // 更新並儲存此角色的對話歷史 chatHistory.push(userMsg, { role: "model", parts: [{ text: aiText }] }); if(chatHistory.length > 20) chatHistory = chatHistory.slice(-20); characters[currentId].history = chatHistory; saveToLocalStorage(); } catch (e) { addBubble("連線異常喵。", 'a'); } } function addBubble(t, type, scroll = true) { const box = document.getElementById('chat-box'); const div = document.createElement('div'); div.className = `msg ${type}`; div.innerText = t; box.appendChild(div); if(scroll) { setTimeout(() => { box.scrollTop = box.scrollHeight; }, 50); } } window.onload = async () => { loadSettings(); renderMessages(); await initModel(); };
